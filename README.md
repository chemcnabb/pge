# pge
site build with broken receiver

***********
Quick Start
***********
	$ clone this locally
	$ cd into pge/
    $ pip install --upgrade virtualenv
    $ virtualenv venv
    $ source venv/bin/activate
    (env) $ pip install djangocms-installer
    (env) $ mkdir pge-cms && cd pge-cms
    (env) $ djangocms -f -p . cmsproject
    (env) $ python manage.py
